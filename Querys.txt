//Query para verificar gastos do mês atual de cada cliente
SELECT CLIENTES.ID_CLIENTE, SUM(TRANSACOES.VL_TRANSACAO)
FROM CLIENTES
INNER JOIN TRANSACOES ON CLIENTES.ID_CLIENTE=TRANSACOES.ID_CLIENTE
WHERE MONTH(DT_TRANSACAO) = "MES ATUAL"
GROUP BY(CLIENTES.ID_CLIENTE)


//Query para verificar faturamento do mês anterior das lojas
SELECT LOJAS.ID_LOJA, SUM(TRANSACOES.VL_TRANSACAO)
FROM LOJAS
INNER JOIN TRANSACOES ON LOJAS.ID_LOJA = TRANSACOES.ID_LOJA
WHERE MONTH(DT_TRANSACAO) = "MES ANTERIOR"
GROUP BY(LOJAS.ID_LOJA)
