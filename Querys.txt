SELECT CLIENTES.ID_CLIENTE, CLIENTES.NM_CLIENTE, TRANSACOES.DT_TRANSACAO,TRANSACOES.VL_TRANSACAO
FROM CLIENTES
INNER JOIN TRANSACOES ON CLIENTES.ID_CLIENTE=TRANSACOES.ID_CLIENTE
SUM(VL_TRANSACAO)
GROUP BY (ID_CLIENTE)
WHERE MONTH(DT_TRANSACAO) = "MES ATUAL"

SELECT LOJAS.ID_LOJA, LOJAS.NM_LOJA, TRANSACOES.VL_TRANSACAO, TRANSACOES.DT_TRANSACAO
FROM LOJAS
INNER JOIN TRANSACOES ON LOJAS.ID_LOJA=TRANSACOES.ID_LOJA
SUM(VL_TRANSACAO)
GROUP BY(ID_LOJA)
WHERE MONTH(DT_TRANSAO)= "MES ANTERIOR"
